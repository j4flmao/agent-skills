# Rust FFI and C Bindings

## FFI Basics

### Foreign Function Interface
```rust
extern "C" {
    fn strlen(s: *const c_char) -> usize;
    fn puts(s: *const c_char) -> c_int;
}

pub fn safe_strlen(s: &str) -> usize {
    let c_str = CString::new(s).expect("CString conversion failed");
    unsafe { strlen(c_str.as_ptr()) }
}
```

## C ABI Functions

### Exposing Rust to C
```rust
use std::ffi::{CStr, CString};
use std::os::raw::c_char;

#[repr(C)]
pub struct User {
    id: i64,
    name: *mut c_char,
    email: *mut c_char,
}

#[no_mangle]
pub extern "C" fn create_user(id: i64, name: *const c_char, email: *const c_char) -> *mut User {
    let name_str = unsafe { CStr::from_ptr(name) }.to_str().unwrap();
    let email_str = unsafe { CStr::from_ptr(email) }.to_str().unwrap();

    let user = Box::new(User {
        id,
        name: CString::new(name_str).unwrap().into_raw(),
        email: CString::new(email_str).unwrap().into_raw(),
    });

    Box::into_raw(user)
}

#[no_mangle]
pub extern "C" fn free_user(user: *mut User) {
    if !user.is_null() {
        unsafe {
            let user = Box::from_raw(user);
            let _ = CString::from_raw(user.name);
            let _ = CString::from_raw(user.email);
        }
    }
}

#[no_mangle]
pub extern "C" fn get_user_id(user: *const User) -> i64 {
    unsafe { (*user).id }
}
```

## Safe Wrappers

### Wrapping C Libraries
```rust
use std::ffi::{CStr, CString};
use std::marker::PhantomData;
use std::ops::Drop;

mod ffi {
    use std::os::raw::c_char;

    extern "C" {
        pub fn sqlite3_open(filename: *const c_char, ppDb: *mut *mut sqlite3) -> c_int;
        pub fn sqlite3_close(db: *mut sqlite3) -> c_int;
        pub fn sqlite3_exec(
            db: *mut sqlite3,
            sql: *const c_char,
            callback: Option<extern "C" fn(*mut c_void, c_int, *mut *mut c_char, *mut *mut c_char) -> c_int>,
            arg: *mut c_void,
            errmsg: *mut *mut c_char,
        ) -> c_int;
    }

    pub enum sqlite3 {}
}

pub struct Sqlite3 {
    db: *mut ffi::sqlite3,
}

impl Sqlite3 {
    pub fn open(path: &str) -> Result<Self, SqliteError> {
        let c_path = CString::new(path).map_err(|_| SqliteError::InvalidPath)?;
        let mut db: *mut ffi::sqlite3 = std::ptr::null_mut();

        let rc = unsafe { ffi::sqlite3_open(c_path.as_ptr(), &mut db) };
        if rc != 0 {
            return Err(SqliteError::OpenFailed(rc));
        }

        Ok(Self { db })
    }

    pub fn execute(&self, sql: &str) -> Result<(), SqliteError> {
        let c_sql = CString::new(sql).map_err(|_| SqliteError::InvalidSql)?;
        let mut errmsg: *mut c_char = std::ptr::null_mut();

        let rc = unsafe {
            ffi::sqlite3_exec(self.db, c_sql.as_ptr(), None, std::ptr::null_mut(), &mut errmsg)
        };

        if rc != 0 {
            let error = if !errmsg.is_null() {
                unsafe { CStr::from_ptr(errmsg) }.to_str().unwrap_or("Unknown").to_string()
            } else {
                format!("Error code: {}", rc)
            };
            return Err(SqliteError::ExecutionError(error));
        }

        Ok(())
    }
}

impl Drop for Sqlite3 {
    fn drop(&mut self) {
        if !self.db.is_null() {
            unsafe { ffi::sqlite3_close(self.db) };
        }
    }
}
```

## Error Handling

### FFI Error Translation
```rust
use std::ffi::NulError;
use std::error::Error;
use std::fmt;

#[derive(Debug)]
pub enum FfiError {
    NulError(NulError),
    NullPointer(&'static str),
    InvalidUtf8(std::str::Utf8Error),
    LibraryError(String),
}

impl fmt::Display for FfiError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            FfiError::NulError(e) => write!(f, "Null byte in string: {}", e),
            FfiError::NullPointer(name) => write!(f, "Null pointer returned for {}", name),
            FfiError::InvalidUtf8(e) => write!(f, "Invalid UTF-8: {}", e),
            FfiError::LibraryError(e) => write!(f, "Library error: {}", e),
        }
    }
}

impl Error for FfiError {}
```

## Memory Management

### Ownership Patterns
```rust
// Option 1: Rust owns, C borrows
pub fn process_data(data: &[u8]) -> i32 {
    let len = data.len();
    let ptr = data.as_ptr();
    unsafe { c_process_data(ptr, len) }
}

// Option 2: C owns, Rust borrows (zero-copy)
pub fn process_c_string(ptr: *const c_char) -> Result<&str, FfiError> {
    if ptr.is_null() {
        return Err(FfiError::NullPointer("c_string"));
    }
    let c_str = unsafe { CStr::from_ptr(ptr) };
    c_str.to_str().map_err(FfiError::InvalidUtf8)
}

// Option 3: Rust allocates, transfers ownership to C
pub fn allocate_buffer(size: usize) -> *mut u8 {
    let mut vec = vec![0u8; size];
    let ptr = vec.as_mut_ptr();
    std::mem::forget(vec); // Prevent Rust from freeing
    ptr
}

// Option 4: C allocates, Rust owns (must free in Rust)
pub fn get_string_from_c() -> Result<String, FfiError> {
    let ptr = unsafe { c_get_string() };
    if ptr.is_null() {
        return Err(FfiError::NullPointer("c_get_string"));
    }
    let c_str = unsafe { CStr::from_ptr(ptr) };
    let result = c_str.to_str().map_err(FfiError::InvalidUtf8)?.to_owned();
    unsafe { c_free_string(ptr) }; // Free C allocation
    Ok(result)
}
```

## Build Configuration

### build.rs
```rust
// build.rs
fn main() {
    // Link to C library
    println!("cargo:rustc-link-lib=my_c_lib");
    println!("cargo:rustc-link-search=/usr/local/lib");

    // Re-run if C headers change
    println!("cargo:rerun-if-changed=include/my_lib.h");

    // Generate bindings
    let bindings = bindgen::Builder::default()
        .header("include/my_lib.h")
        .generate()
        .expect("Failed to generate bindings");

    bindings
        .write_to_file("src/bindings.rs")
        .expect("Failed to write bindings");
}
```

## Key Points
- extern "C" defines Rust functions callable from other languages
- #[no_mangle] prevents name mangling for C symbol resolution
- CString and CStr handle null-terminated string conversions
- Box::into_raw transfers ownership to C, Box::from_raw reclaims it
- Always validate pointers for null before dereferencing in FFI
- Safe wrappers encapsulate unsafe FFI calls behind Rust-safe APIs
- Drop implementations ensure C resources are properly freed
- cbindgen auto-generates C headers from Rust FFI exports
- build.rs with bindgen generates Rust bindings from C headers
- Error translation converts C error codes to Rust Result types
